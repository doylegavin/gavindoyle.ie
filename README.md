## Component: Live website preview (with graceful fallback)

> Many sites block iframes via `X-Frame-Options`. This component attempts an iframe and falls back to a screenshot + link.

```tsx
// src/components/WebsitePreview.tsx
'use client'
import Image from 'next/image'
import { useState } from 'react'

type Props = { title: string; url: string; img: string } // img is a local screenshot

export default function WebsitePreview({ title, url, img }: Props) {
  const [blocked, setBlocked] = useState(false)
  return (
    <div className="rounded-2xl border overflow-hidden shadow-sm">
      {!blocked ? (
        <iframe
          src={url}
          title={title}
          onError={() => setBlocked(true)}
          className="w-full h-[540px] bg-white"
          loading="lazy"
        />
      ) : (
        <div className="relative w-full h-[320px]">
          <Image src={img} alt={`${title} preview`} fill className="object-cover" />
        </div>
      )}
      <div className="p-4 flex items-center justify-between">
        <h3 className="font-semibold">{title}</h3>
        <a className="underline" href={url} target="_blank" rel="noreferrer">Open site</a>
      </div>
    </div>
  )
}
```

**Tip:** Pre‑generate screenshots with a simple Puppeteer script during build, or just drop static JPGs in `/public/previews/` as `examinaite.jpg`, `dronan.jpg`, `kneecapp.jpg`, `gaeilgeoir.jpg`.
